1use crate::protocol::AAMNPacket;
2use anyhow::{anyhow, Result};
3use quinn::{ClientConfig, Connection, Endpoint, ServerConfig};
4use rustls::{Certificate, PrivateKey, RootCertStore};
5use std::net::SocketAddr;

9pub struct TransportLayer {
10    endpoint: Endpoint,
11    #[allow(dead_code)]
12    root_store: RootCertStore,
13    known_node_certs: Arc<Mutex<Vec<Certificate>>>,
14    server_certificate: Certificate,
15}
16
17impl TransportLayer {
18    pub fn new(bind_addr: SocketAddr) -> Result<Self> {
19        let server_config = Self::make_server_config()?;
20        let endpoint = Endpoint::server(server_config, bind_addr)?;
21        let root_store = RootCertStore::empty();
22        let cert = rcgen::generate_simple_self_signed(vec!["aamn-service".into()])?;
23        let server_certificate = Certificate(cert.serialize_der()?);
24        Ok(Self {
25            endpoint,
26            root_store,
27            known_node_certs: Arc::new(Mutex::new(Vec::new())),
28            server_certificate,
29        })
30    }
31
32    pub async fn connect(&self, addr: SocketAddr) -> Result<Connection> {
33        let client_config = self.make_client_config()?;
34        let connection = self
35            .endpoint
36            .connect_with(client_config, addr, "aamn")
37            .map_err(|e| anyhow!("Error: {}", e))?
38            .await
39            .map_err(|e| anyhow!("Fallo: {}", e))?;
40        Ok(connection)
41    }
42
43    pub async fn connect_with_verification(
44        &self,
45        addr: SocketAddr,
46        _expected_node_id: &[u8; 32],
47    ) -> Result<Connection> {
48        // En una versión futura, este método verificará que el certificado
49        // remoto esté vinculado a la identidad esperada del nodo. Por ahora,
50        // reutilizamos `connect` pero mantenemos la firma para no romper la API.
51        self.connect(addr).await
52    }
53
54    fn make_client_config(&self) -> Result<ClientConfig> {
55        // En lugar de confiar en el almacén de roots del sistema, limitamos
56        // explícitamente la confianza a los certificados conocidos si están
57        // configurados, manteniendo compatibilidad hacia atrás.
58        let mut roots = self.root_store.clone();
59        {
60            let known = self.known_node_certs.lock().map_err(|_| anyhow!("Lock"))?;
61            for cert in known.iter() {
62                roots.add(cert).map_err(|_| anyhow!("Invalid cert"))?;
63            }
64        }
65
66        let client_config = ClientConfig::with_root_certificates(roots);
67        Ok(client_config)
68    }
69
70    pub async fn send_packet(&self, connection: &Connection, packet: &AAMNPacket) -> Result<()> {
71        let mut send_stream = connection.open_uni().await?;
72        send_stream.write_all(&bincode::serialize(packet)?).await?;
73        send_stream.finish().await?;
74        Ok(())
75    }
76
77    pub async fn listen_packets<F>(&self, handler: F) -> Result<()>
78    where
79        F: Fn(AAMNPacket) + Send + Sync + 'static,
80    {
81        let handler = Arc::new(handler);
82        while let Some(conn) = self.endpoint.accept().await {
83            let h = Arc::clone(&handler);
84            tokio::spawn(async move {
85                if let Ok(connection) = conn.await {
86                    while let Ok(mut recv) = connection.accept_uni().await {
87                        let mut buf = vec![0u8; 65535];
88                        if let Ok(Some(n)) = recv.read(&mut buf).await {
89                            if let Ok(p) = bincode::deserialize::<AAMNPacket>(&buf[..n]) {
90                                h(p);
91                            }
92                        }
93                    }
94                }
95            });
96        }
97        Ok(())
98    }
99
100    fn make_server_config() -> Result<ServerConfig> {
101        let cert = rcgen::generate_simple_self_signed(vec!["aamn-service".into()])?;
102        Ok(ServerConfig::with_single_cert(
103            vec![Certificate(cert.serialize_der()?)],
104            PrivateKey(cert.serialize_private_key_der()),
105        )?)
106    }
107
108    pub fn add_known_node_cert(&self, cert: Certificate) -> Result<()> {
109        self.known_node_certs
110            .lock()
111            .map_err(|_| anyhow!("Lock"))?
112            .push(cert);
113        Ok(())
114    }
115    pub fn get_server_certificate(&self) -> &Certificate {
116        &self.server_certificate
117    }
