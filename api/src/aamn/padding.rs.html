3//! Este módulo implementa protección contra análisis de tráfico usando:
4//! - Celdas de tamaño fijo (como Tor)

8use anyhow::{anyhow, Result};
9use rand::{rngs::OsRng, Rng};
10use std::time::Duration;
11use tokio::sync::mpsc;
12use tokio::time::{interval_at, Instant};

14/// Tamaño fijo de celda (similar a Tor's 512 bytes)
15pub const CELL_SIZE: usize = 512;
16
17/// Tamaño del header de celda
18pub const CELL_HEADER_SIZE: usize = 16;
19
20/// Payload máximo por celda
21pub const CELL_PAYLOAD_SIZE: usize = CELL_SIZE - CELL_HEADER_SIZE;
22
23/// Tipo de mensaje en la celda
24#[derive(Debug, Clone, Copy, PartialEq, Eq)]
25#[repr(u8)]
26pub enum CellType {
27    Data = 0,
28    Padding = 1,
29    Create = 2,
30    Created = 3,
31    Relay = 4,
32    Destroy = 5,
33}
34
35/// Header de celda fijo
36#[derive(Clone)]
37pub struct CellHeader {
38    /// Identificador de circuito
39    pub circuit_id: u32,
40    /// Tipo de mensaje
41    pub cell_type: CellType,
42    /// Longitud del payload
43    pub payload_length: u16,
44    /// Comando específico
45    pub command: u8,
46}
47
48impl CellHeader {
49    /// Serializa el header a bytes
50    pub fn to_bytes(&self) -> [u8; CELL_HEADER_SIZE] {
51        let mut bytes = [0u8; CELL_HEADER_SIZE];
52        bytes[0..4].copy_from_slice(&self.circuit_id.to_le_bytes());
53        bytes[4] = self.cell_type as u8;
54        bytes[5..7].copy_from_slice(&self.payload_length.to_le_bytes());
55        bytes[7] = self.command;
56        bytes
57    }
58
59    /// Deserializa el header desde bytes
60    pub fn from_bytes(bytes: &[u8; CELL_HEADER_SIZE]) -> Result<Self> {
61        let circuit_id = u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]);
62        let cell_type = match bytes[4] {
63            0 => CellType::Data,
64            1 => CellType::Padding,
65            2 => CellType::Create,
66            3 => CellType::Created,
67            4 => CellType::Relay,
68            5 => CellType::Destroy,
69            _ => return Err(anyhow!("Tipo de celda inválido")),
70        };
71        let payload_length = u16::from_le_bytes([bytes[5], bytes[6]]);
72        let command = bytes[7];
73
74        Ok(Self {
75            circuit_id,
76            cell_type,
77            payload_length,
78            command,
79        })
80    }
81}

83/// Celda de tamaño fijo
84#[derive(Clone)]
85pub struct Cell {
86    pub header: CellHeader,
87    pub payload: Vec<u8>,
88}
89
90impl Cell {
91    /// Crea una nueva celda
92    pub fn new(
93        circuit_id: u32,
94        cell_type: CellType,
95        command: u8,
96        payload: Vec<u8>,
97    ) -> Result<Self> {
98        if payload.len() > CELL_PAYLOAD_SIZE {
99            return Err(anyhow!("Payload demasiado grande"));
100        }
101
102        let payload_length = payload.len() as u16;
103
104        Ok(Self {
105            header: CellHeader {
106                circuit_id,
107                cell_type,
108                payload_length,
109                command,
110            },
111            payload,
112        })
113    }
114
115    /// Serializa la celda completa
116    pub fn to_bytes(&self) -> Vec<u8> {
117        let mut result = Vec::with_capacity(CELL_SIZE);
118        result.extend_from_slice(&self.header.to_bytes());
119
120        // Agregar payload
121        result.extend_from_slice(&self.payload);
122
123        // Padding al tamaño fijo
124        let padding_needed = CELL_PAYLOAD_SIZE - self.payload.len();
125        if padding_needed > 0 {
126            let mut padding = vec![0u8; padding_needed];
127            // Padding aleatorio para prevenir análisis
128            let mut rng = OsRng;
129            rng.fill(&mut padding[..]);
130            result.extend_from_slice(&padding);
131        }
132
133        result
134    }
135
136    /// Deserializa una celda
137    pub fn from_bytes(data: &[u8]) -> Result<Self> {
138        if data.len() < CELL_SIZE {
139            return Err(anyhow!("Datos insuficientes para celda"));
140        }
141
142        let header_bytes: &[u8; CELL_HEADER_SIZE] = data[0..CELL_HEADER_SIZE]
143            .try_into()
144            .map_err(|_| anyhow!("Header inválido"))?;
145
146        let header = CellHeader::from_bytes(header_bytes)?;
147
148        // Extraer payload (sin padding)
149        let payload_end = CELL_HEADER_SIZE + header.payload_length as usize;
150        let payload = data[CELL_HEADER_SIZE..payload_end].to_vec();
151
152        Ok(Self { header, payload })
153    }
154
155    /// Verifica si la celda es padding
156    pub fn is_padding(&self) -> bool {
157        self.header.cell_type == CellType::Padding
158    }
159}

161/// Gestor de padding para protección contra análisis de tráfico
162pub struct TrafficPadding {
163    /// Probabilidad de enviar celdas de padding (0.0 - 1.0)
164    padding_probability: f32,
165    /// Intervalo mínimo entre celdas de padding
166    min_interval_ms: u64,
167    /// Tamaño objetivo para padding de longitud
168    target_size: usize,
169}
170
171impl TrafficPadding {
172    /// Crea un nuevo gestor de padding
173    pub fn new() -> Self {
174        Self {
175            padding_probability: 0.1, // 10% de probabilidad
176            min_interval_ms: 100,
177            target_size: CELL_PAYLOAD_SIZE,
178        }
179    }
180
181    /// Configura la probabilidad de padding
182    pub fn with_probability(mut self, prob: f32) -> Self {
183        self.padding_probability = prob.clamp(0.0, 1.0);
184        self
185    }
186
187    /// Configura el intervalo mínimo
188    pub fn with_min_interval(mut self, ms: u64) -> Self {
189        self.min_interval_ms = ms;
190        self
191    }
192
193    /// Aplica padding al payload para ocultar el tamaño real
194    pub fn apply_length_padding(&self, data: &[u8]) -> Vec<u8> {
195        let mut rng = OsRng;
196
197        // Estrategia: añadir padding aleatorio para alcanzar tamaño fijo
198        // Esto previene ataques de correlación de longitud
199        if data.len() < self.target_size {
200            let mut result = data.to_vec();
201            let padding_len = self.target_size - data.len();
202
203            // Padding aleatorio
204            let mut padding = vec![0u8; padding_len];
205            rng.fill(&mut padding[..]);
206
207            // Marcador de inicio de padding (para que el receptor pueda distinguir)
208            // Usamos el último byte como longitud real
209            result.push(data.len() as u8);
210            result.extend_from_slice(&padding[..padding_len - 1]);
211
212            result
213        } else if data.len() > self.target_size {
214            // Dividir en múltiples celdas
215            data.to_vec()
216        } else {
217            data.to_vec()
218        }
219    }
220
221    /// Determina si se debe enviar una celda de padding
222    pub fn should_send_padding(&self) -> bool {
223        let mut rng = OsRng;
224        rng.gen::<f32>() < self.padding_probability
225    }
226
227    /// Genera una celda de padding
228    pub fn generate_padding_cell(&self, circuit_id: u32) -> Cell {
229        let mut rng = OsRng;
230        let mut payload = vec![0u8; CELL_PAYLOAD_SIZE];
231        rng.fill(&mut payload[..]);
232
233        Cell::new(circuit_id, CellType::Padding, 0, payload).unwrap()
234    }
235
236    /// Ajusta el padding basado en el tráfico entrante (para simetría)
237    pub fn adjust_for_incoming_traffic(&mut self, incoming_size: usize) {
238        // Si hay mucho tráfico entrante, reducir padding saliente
239        // Esto crea un patrón más natural
240        if incoming_size > 10000 {
241            self.padding_probability = 0.05;
242        } else if incoming_size > 5000 {
243            self.padding_probability = 0.1;
244        } else {
245            self.padding_probability = 0.15;
246        }
247    }
248}

250impl Default for TrafficPadding {
251    fn default() -> Self {
252        Self::new()
253    }

255///
256/// Emite paquetes a un ritmo constante configurado.
257/// Si hay paquetes reales para enviar, se priorizan.
258/// Si no, emite paquetes de padding para rellenar el ancho de banda y mitigar análisis estadístico.
259pub struct CoverTrafficManager {
260    /// Tasa de envío objetivo (celdas por segundo)
261    cells_per_second: u32,
262    /// Generador de padding activo
263    generator_active: bool,
264}
265
266impl CoverTrafficManager {
267    /// Inicializa un gestor de tráfico constante
268    pub fn new(cells_per_second: u32) -> Self {
269        Self {
270            cells_per_second,
271            generator_active: false,
272        }
273    }
274
275    /// Inicia un bucle asíncrono que emite tráfico de cover a ritmo constante.
276    /// Retorna un channel por donde se pueden recibir las celdas a enviar por la red.
277    pub fn start_generator(&mut self) -> mpsc::Receiver<Cell> {
278        let (tx, rx) = mpsc::channel(100);
279        let cps = self.cells_per_second.max(1);
280        let duration = Duration::from_millis(1000 / cps as u64);
281
282        self.generator_active = true;
283
284        tokio::spawn(async move {
285            let start = Instant::now() + duration;
286            let mut interval = interval_at(start, duration);
287
288            loop {
289                interval.tick().await;
290
291                // Generar celda de padding (circuit_id 0 = dummy padding)
292                let mut payload = vec![0u8; CELL_PAYLOAD_SIZE];
293                rand::thread_rng().fill(&mut payload[..]);
294
295                if let Ok(cell) = Cell::new(0, CellType::Padding, 0, payload) {
296                    // Si el canal está lleno o cerrado, ignoramos
297                    if tx.try_send(cell).is_err() {
298                        // Receptor desconectado o buffer lleno (posible congestión)
299                    }
300                }
301            }
302        });
303
304        rx
305    }
306
307    pub fn is_active(&self) -> bool {
308        self.generator_active
309    }
310}
311
312impl Default for CoverTrafficManager {
313    fn default() -> Self {
314        Self::new(10) // Por defecto: 10 celdas/sec -> ~5KB/s
315    }
316}

318/// Gestor de traffic shaping para prevenir análisis
319pub struct TrafficShaper {
320    /// Ancho de banda objetivo (bytes por segundo)
321    target_bandwidth: u64,
322    /// Burst máximo
323    max_burst: u64,
324    /// Tokens actuales
325    tokens: u64,
326    /// Última actualización
327    last_update: std::time::Instant,
328}
329
330impl TrafficShaper {
331    /// Crea un nuevo shaper
332    pub fn new(target_bps: u64, max_burst: u64) -> Self {
333        Self {
334            target_bandwidth: target_bps,
335            max_burst,
336            tokens: max_burst,
337            last_update: std::time::Instant::now(),
338        }
339    }
340
341    /// Verifica si se puede enviar datos
342    pub fn can_send(&mut self, size: usize) -> bool {
343        self.refill();
344        self.tokens >= size as u64
345    }
346
347    /// Envía datos (deduce tokens)
348    pub fn send(&mut self, size: usize) {
349        self.refill();
350        if self.tokens >= size as u64 {
351            self.tokens -= size as u64;
352        }
353    }
354
355    /// Rellena los tokens basándose en el tiempo transcurrido
356    fn refill(&mut self) {
357        let elapsed = self.last_update.elapsed().as_secs_f64();
358        let tokens_to_add = (self.target_bandwidth as f64 * elapsed) as u64;
359
360        self.tokens = (self.tokens + tokens_to_add).min(self.max_burst);
361        self.last_update = std::time::Instant::now();
362    }
363}

369    #[test]
370    fn test_cell_serialization() {
371        let payload = b"Test message".to_vec();
372        let cell = Cell::new(1, CellType::Data, 0, payload.clone()).unwrap();
373
374        let bytes = cell.to_bytes();
375        assert_eq!(bytes.len(), CELL_SIZE);
376
377        let decoded = Cell::from_bytes(&bytes).unwrap();
378        assert_eq!(decoded.header.circuit_id, 1);
379        assert_eq!(decoded.header.cell_type, CellType::Data);
380        assert_eq!(decoded.payload, payload);
381    }

385        let padding = TrafficPadding::new();
386        let small_data = b"Hello".to_vec();
387

389        // El resultado debe ser mayor o igual al target
